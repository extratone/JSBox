You can use a wrapper function. I generally do something like this:

// create a wrapper function that returns a boolean
let f = () => {
    let p = new Prompt();
    // configure prompt
    if (!p.show()) {
        return false;
    }

    let p2 = new Prompt();
    // configure prompt
    if (!p2.show()) {
        return false;
    }

    // we have all the info, do the work!
    return true;
}

// call the wrapper, use result to control cancellation
if (!f()) {
    context.cancel();
}
