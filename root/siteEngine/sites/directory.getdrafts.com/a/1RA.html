Write Good

Posted by @fffergal, Last update about 3 years ago

Use “write good” by Brian Ford to make readability suggestions.

“write good” javascript can be found at https://github.com/btford/write-good.

Works great when paired with the “Readability Scores” action.

11 April 2019 Now you can tap on a highlight in the body to see it in the suggestion list, and tap the highlight in the suggestion list to go back to the body.

var writeGoodDrafts = function() {
	var suggestions = writeGood(draft.content);

	var suggestionParts = [];
	suggestions.forEach(function(suggestion, index) {
		var startLine = draft.content.lastIndexOf('\n', suggestion.index) + 1;
		if (suggestion.index - startLine < 25) {
			var contextStart = startLine;
		}
		else {
			var contextStart = suggestion.index - 25;
		}
		var nextLine = draft.content.indexOf('\n', suggestion.index + suggestion.offset);
		if (nextLine == -1) {
			nextLine = draft.content.length;
		}
		var context = '' + draft.content.substring(contextStart, nextLine).substr(0, 70) + '';
		var contextIndex = suggestion.index - contextStart + 1;
		var highlightSpec = {'index': contextIndex, 'offset': suggestion.offset};
		var contextMD = writeGoodHighlight(context, [highlightSpec], 'inplace', 'suggestionHighlight', index);
		suggestionParts.push({'context': contextMD, 'reason': suggestion.reason});
	});
	var suggestionItems = [];
	suggestionParts.forEach(function(suggestionPart, index) {
		var suggestionItemMD = '<li id="suggestion' + index + '">' + suggestionPart.reason + '<blockquote>' + suggestionPart.context + '</blockquote></li>';
		suggestionItems.push(suggestionItemMD);
	});
	var suggestionsMD = '<ol>' + suggestionItems.join('\n') + '</ol>';

	draft.setTemplateTag('writeGoodSuggestions', suggestionsMD);

	var highlightMD = writeGoodHighlight(draft.content, suggestions);

	draft.setTemplateTag('writeGoodHighlighted', highlightMD);
};

var writeGoodHighlight = function(text, highlights, gotoPrefix = 'suggestion', idPrefix = 'inplace', indexOffset = 0) {
	var highlightPieces = [];
	var highlightStart = 0;
	highlights.forEach(function(suggestion, index) {		highlightPieces.push(text.substring(highlightStart, suggestion.index));
		currentIndex = index + indexOffset;
		highlightPieces.push('<span class="writeGoodHighlight" id="' + idPrefix + currentIndex + '">');
		highlightPieces.push('<a href="#' + gotoPrefix + currentIndex + '">');
		highlightPieces.push(text.substr(suggestion.index, suggestion.offset));
		highlightPieces.push('</a>');
		highlightPieces.push('</span>');
		highlightStart = suggestion.index + suggestion.offset;
	});
	highlightPieces.push(text.substr(highlightStart));
	return highlightPieces.join('');
};
