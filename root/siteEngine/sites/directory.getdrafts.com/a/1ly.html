Action

Emojify

Posted by agiletortoise, Last update about 1 year ago

Swap common words with their matching emoji characters, or reverse the process.

Emoji matches are defined in a variable in the first step of the action, and can be easily modified to support other emoji. Only a few emoji are included as examples, like ‚Äúpizza‚Äù, ‚Äúdog‚Äù, ‚Äúcat‚Äù, ‚Äúshrug‚Äù. Add addition ones by editing the ‚Äúemojis‚Äù variable with other values.

The action also supports the option to require the word be followed by the word ‚Äúemoji‚Äù when the ‚ÄúRequire ‚Äòemoji‚Äô keyword is enabled. This works well with dictation to allow you to say, ‚ÄúDo you want to eat pizza emoji?‚Äù and later convert only the words you marked with ‚Äúemoji‚Äù.

    function inverse(obj){
  var retobj = {};
  for(var key in obj){
    retobj[obj[key]] = key;
  }
  return retobj;
}

let f = () => {
	let p = new Prompt();
	p.title = "Emojify";
	p.message = "Would you like to emojify (convert text to emoji) or de-emojify (replace emoji with text)?"
	p.addSwitch("requireEmoji", "Require \"emoji\" after", requireEmojiKeyword);
	p.addButton("Emojify");
	p.addButton("De-Emojify");

	if (!p.show()) { return  false; }

	requireEmojiKeyword = p.fieldValues['requireEmoji'];
	let [st, len] = editor.getSelectedRange();
	if (len == 0) { // no selection, use full draft
		st = 0;
		len = editor.getText().length;
	}
	let text = editor.getTextInRange(st, len);

	const isReversing = p.buttonPressed == 'De-Emojify';
	const vals = (isReversing ? inverse(emojis) : emojis);
	for (let key in vals) {
		let lookup = key;
		if (!isReversing && requireEmojiKeyword) {
			lookup = `${key} emoji`;
		}
		let re = new RegExp(`${lookup}`, 'gi');
		text = text.replace(re, vals[key]);
	}
	editor.setTextInRange(st, len, text);
}

if (!f()) {
	context.cancel();
}
